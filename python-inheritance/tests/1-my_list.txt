# Importing function from the module:
    >>> MyList = __import__('1-my_list').MyList

# Testing if the function correctly sorts and prints a list of positive integers.
    >>> my_list = MyList()
    >>> my_list.append(5)
    >>> my_list.append(1)
    >>> my_list.append(3)
    >>> my_list.append(4)
    >>> my_list.append(2)
    >>> my_list.print_sorted()
    [1, 2, 3, 4, 5]


# Testing if the function correctly sorts and prints a list of negative integers.
    >>> my_list = MyList()
    >>> my_list.append(-5)
    >>> my_list.append(-1)
    >>> my_list.append(-3)
    >>> my_list.append(-4)
    >>> my_list.append(-2)
    >>> my_list.print_sorted()
    [-5, -4, -3, -2, -1]


# Testing if the function correctly sorts and prints a list of mixed positive and negative integers.
    >>> my_list = MyList()
    >>> my_list.append(-5)
    >>> my_list.append(1)
    >>> my_list.append(-3)
    >>> my_list.append(4)
    >>> my_list.append(-2)
    >>> my_list.print_sorted()
    [-5, -3, -2, 1, 4]


# Testing if the function correctly sorts and prints a list of duplicate integers.
    >>> my_list = MyList()
    >>> my_list.append(5)
    >>> my_list.append(1)
    >>> my_list.append(3)
    >>> my_list.append(4)
    >>> my_list.append(2)
    >>> my_list.append(4)
    >>> my_list.append(1)
    >>> my_list.append(2)
    >>> my_list.print_sorted()
    [1, 1, 2, 2, 3, 4, 4, 5]
